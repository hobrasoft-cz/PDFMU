17  package cz.hobrasoft.pdfmu;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.List;
23  import org.junit.Assert;
24  import org.junit.Test;
25  import org.junit.contrib.java.lang.system.Assertion;
26  
27  /**
28   * Since {@link cz.hobrasoft.pdfmu.operation.OperationInspect} does not inspect
29   * the attachments, we cannot test the {@code attach} operation properly. We
30   * cannot compare the result bit-by-bit either because it differs in ModDate. We
31   * only test whether a single basic call succeeds.
32   *
33   * @author Filip Bartek
34   */
35  public class MainAttachTest extends MainTest {
36  
37      @Test
38      public void testAttach() throws IOException {
39          List<String> argsList = new ArrayList<>();
40          argsList.add("attach");
41          final File inputFile = BLANK_12_PDF.getFile(folder);
42          argsList.add(inputFile.getAbsolutePath());
43          FileResource attachmentFileResource = new FileResource("blank.txt");
44          File attachmentFile = attachmentFileResource.getFile(folder);
45          argsList.add(attachmentFile.getAbsolutePath());
46          argsList.add("--out");
47          final File outFile = newFile("out.pdf", false);
48          argsList.add(outFile.getAbsolutePath());
49          exit.expectSystemExitWithStatus(0);
50          exit.checkAssertionAfterwards(new Assertion() {
51              @Override
52              public void checkAssertion() {
53                  Assert.assertTrue(outFile.exists());
54                  Assert.assertTrue(outFile.length() > inputFile.length());
55              }
56          });
57          Main.main(argsList.toArray(new String[]{}));
58          assert false;
59      }
60  }
