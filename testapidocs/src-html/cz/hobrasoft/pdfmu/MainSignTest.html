131    private static void assertDateBetween(Date actual, Date begin, Date end) {
132        assert actual != null;
133        assert begin != null;
134        // The dates saved in PDF files only have second precision,
135        // while new Date() has millisecond precision.
136        // We compare with a 1 second delta to compensate
137        Assert.assertTrue(actual.getTime() - begin.getTime()
138                > TimeUnit.SECONDS.toMillis(-1));
139        assert end != null;
140        Assert.assertFalse(actual.after(end));
141    }

143    // Example content of the properties dictionary:
144    // Producer => iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version); modified using iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version)
145    // CreationDate => D:20160525204745+02'00'
146    // ModDate => D:20160601102240+02'00'
147    private static void assertDefaultPropertiesValid(
148            Map<String, String> properties, String expectedCreationDate,
149            Date modDateBegin, Date modDateEnd) throws ParseException {
150        Assert.assertNotNull(properties);
151        Assert.assertEquals(3, properties.size());
152        Assert.assertTrue(properties.containsKey("Producer"));
153        String expectedProducer = "iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version); modified using iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version)";
154        Assert.assertEquals(expectedProducer, properties.get("Producer"));
155        Assert.assertTrue(properties.containsKey("CreationDate"));
156        assert expectedCreationDate != null;
157        Assert.assertEquals(expectedCreationDate, properties.get("CreationDate"));
158        Assert.assertTrue(properties.containsKey("ModDate"));
159        DateFormat dateFormat = new SimpleDateFormat("'D:'yyyyMMddHHmmssX'''00'''");
160        Date actualModDate = dateFormat.parse(properties.get("ModDate"));
161        assertDateBetween(actualModDate, modDateBegin, modDateEnd);
162    }

164    @Test
165    public void test1p12() throws IOException {
166        final PdfFileResource inFileResource = BLANK_12_PDF;
167        File inFile = inFileResource.getFile(folder);
168        File keystoreFile = new FileResource("1.p12").getFile(folder);
169        final File outFile = newFile("out.pdf", false);
170
171        List<String> argsList = new ArrayList<>();
172        argsList.add("sign");
173        argsList.add(inFile.getAbsolutePath());
174        argsList.add("--out");
175        argsList.add(outFile.getAbsolutePath());
176        argsList.add("--keystore");
177        argsList.add(keystoreFile.getAbsolutePath());
178
179        final Date modDateBegin = new Date();
180
181        exit.expectSystemExitWithStatus(0);
182        exit.checkAssertionAfterwards(new Assertion() {
183            @Override
184            public void checkAssertion() throws OperationException, IOException, ParseException {
185                Date modDateEnd = new Date();
186                Assert.assertTrue(outFile.exists());
187                Inspect inspect = OperationInspect.getInstance().execute(outFile);
188                assert inspect != null;
189                assert inFileResource.version != null;
190                Assert.assertEquals(inFileResource.version, inspect.version);
191                assertDefaultPropertiesValid(inspect.properties,
192                        inFileResource.creationDate, modDateBegin, modDateEnd);
193                Assert.assertNotNull(inspect.signatures);
194                Assert.assertEquals(1, inspect.signatures.nRevisions.intValue());
195                Assert.assertNotNull(inspect.signatures.signatures);
196                Assert.assertEquals(1, inspect.signatures.signatures.size());
197                {
198                    Signature signature = inspect.signatures.signatures.get(0);
199                    Assert.assertNotNull(signature);
200                    Assert.assertEquals("Signature1", signature.id);
201                    Assert.assertTrue(signature.coversWholeDocument);
202                    Assert.assertEquals(1, signature.revision.intValue());
203                    {
204                        SignatureMetadata metadata = signature.metadata;
205                        Assert.assertNotNull(metadata);
206                        Assert.assertNull(metadata.name);
207                        Assert.assertEquals("", metadata.reason);
208                        Assert.assertEquals("", metadata.location);
209                        Assert.assertNotNull(metadata.date);
210                        DateFormat dateFormat = new SimpleDateFormat("EEE MMM dd HH:mm:ss zzz yyyy");
211                        // Example: "Wed Jun 01 11:22:38 CEST 2016"
212                        Date actualDate = dateFormat.parse(metadata.date);
213                        assertDateBetween(actualDate, modDateBegin, modDateEnd);
214                    }
215                    Assert.assertNotNull(signature.certificates);
216                    Assert.assertEquals(1, signature.certificates.size());
217                    {
218                        CertificateResult certificate = signature.certificates.get(0);
219                        Assert.assertNotNull(certificate);
220                        Assert.assertEquals("X.509", certificate.type);
221                        Assert.assertTrue(certificate.selfSigned);
222                        Map<String, List<String>> dn = new LinkedHashMap<>();
223                        dn.put("CN", Arrays.asList(new String[]{"CN1"}));
224                        dn.put("E", Arrays.asList(new String[]{"E1"}));
225                        dn.put("OU", Arrays.asList(new String[]{"OU1"}));
226                        dn.put("O", Arrays.asList(new String[]{"O1"}));
227                        dn.put("L", Arrays.asList(new String[]{"L1"}));
228                        dn.put("ST", Arrays.asList(new String[]{"ST1"}));
229                        dn.put("C", Arrays.asList(new String[]{"C1"}));
230                        Assert.assertEquals(dn, certificate.subject);
231                        Assert.assertEquals(dn, certificate.issuer);
232                    }
233                }
234            }
235        });
236        Main.main(argsList.toArray(new String[]{}));
237        assert false;
238    }
