125    private static void assertDateBetween(Date actual, Date begin, Date end) {
126        assert actual != null;
127        assert begin != null;
128        // The dates saved in PDF files only have second precision,
129        // while new Date() has millisecond precision.
130        // We compare with a 1 second delta to compensate
131        Assert.assertTrue(actual.getTime() - begin.getTime()
132                > TimeUnit.SECONDS.toMillis(-1));
133        assert end != null;
134        Assert.assertFalse(actual.after(end));
135    }

137    // Example content of the properties dictionary:
138    // Producer => iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version); modified using iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version)
139    // CreationDate => D:20160525204745+02'00'
140    // ModDate => D:20160601102240+02'00'
141    private static void assertDefaultPropertiesValid(
142            Map<String, String> properties, String expectedCreationDate,
143            Date modDateBegin, Date modDateEnd) throws ParseException {
144        Assert.assertNotNull(properties);
145        Assert.assertEquals(3, properties.size());
146        Assert.assertTrue(properties.containsKey("Producer"));
147        String expectedProducer = "iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version); modified using iText® 5.5.6 ©2000-2015 iText Group NV (AGPL-version)";
148        Assert.assertEquals(expectedProducer, properties.get("Producer"));
149        Assert.assertTrue(properties.containsKey("CreationDate"));
150        assert expectedCreationDate != null;
151        Assert.assertEquals(expectedCreationDate, properties.get("CreationDate"));
152        Assert.assertTrue(properties.containsKey("ModDate"));
153        DateFormat dateFormat = new SimpleDateFormat("'D:'yyyyMMddHHmmssX'''00'''");
154        Date actualModDate = dateFormat.parse(properties.get("ModDate"));
155        assertDateBetween(actualModDate, modDateBegin, modDateEnd);
156    }

158    @Test
159    public void test1p12() throws IOException {
160        final PdfFileResource inFileResource = BLANK_12_PDF;
161        File inFile = inFileResource.getFile(folder);
162        File keystoreFile = new FileResource("1.p12").getFile(folder);
163        final File outFile = newFile("out.pdf", false);
164
165        List<String> argsList = new ArrayList<>();
166        argsList.add("sign");
167        argsList.add(inFile.getAbsolutePath());
168        argsList.add("--out");
169        argsList.add(outFile.getAbsolutePath());
170        argsList.add("--keystore");
171        argsList.add(keystoreFile.getAbsolutePath());
172
173        final Date modDateBegin = new Date();
174
175        exit.expectSystemExitWithStatus(0);
176        exit.checkAssertionAfterwards(new Assertion() {
177            @Override
178            public void checkAssertion() throws OperationException, IOException, ParseException {
179                Date modDateEnd = new Date();
180                Assert.assertTrue(outFile.exists());
181                Inspect inspect = OperationInspect.getInstance().execute(outFile);
182                assert inspect != null;
183                assert inFileResource.version != null;
184                Assert.assertEquals(inFileResource.version, inspect.version);
185                assertDefaultPropertiesValid(inspect.properties,
186                        inFileResource.creationDate, modDateBegin, modDateEnd);
187                Assert.assertNotNull(inspect.signatures);
188                Assert.assertEquals(1, inspect.signatures.nRevisions.intValue());
189                Assert.assertNotNull(inspect.signatures.signatures);
190                Assert.assertEquals(1, inspect.signatures.signatures.size());
191                {
192                    Signature signature = inspect.signatures.signatures.get(0);
193                    Assert.assertNotNull(signature);
194                    Assert.assertEquals("Signature1", signature.id);
195                    Assert.assertTrue(signature.coversWholeDocument);
196                    Assert.assertEquals(1, signature.revision.intValue());
197                    {
198                        SignatureMetadata metadata = signature.metadata;
199                        Assert.assertNotNull(metadata);
200                        Assert.assertNull(metadata.name);
201                        Assert.assertEquals("", metadata.reason);
202                        Assert.assertEquals("", metadata.location);
203                        Assert.assertNotNull(metadata.date);
204                        DateFormat dateFormat = new SimpleDateFormat("EEE MMM dd HH:mm:ss zzz yyyy");
205                        // Example: "Wed Jun 01 11:22:38 CEST 2016"
206                        Date actualDate = dateFormat.parse(metadata.date);
207                        assertDateBetween(actualDate, modDateBegin, modDateEnd);
208                    }
209                    Assert.assertNotNull(signature.certificates);
210                    Assert.assertEquals(1, signature.certificates.size());
211                    {
212                        CertificateResult certificate = signature.certificates.get(0);
213                        Assert.assertNotNull(certificate);
214                        Assert.assertEquals("X.509", certificate.type);
215                        Assert.assertTrue(certificate.selfSigned);
216                        Map<String, List<String>> dn = new LinkedHashMap<>();
217                        dn.put("CN", Arrays.asList(new String[]{"CN1"}));
218                        dn.put("E", Arrays.asList(new String[]{"E1"}));
219                        dn.put("OU", Arrays.asList(new String[]{"OU1"}));
220                        dn.put("O", Arrays.asList(new String[]{"O1"}));
221                        dn.put("L", Arrays.asList(new String[]{"L1"}));
222                        dn.put("ST", Arrays.asList(new String[]{"ST1"}));
223                        dn.put("C", Arrays.asList(new String[]{"C1"}));
224                        Assert.assertEquals(dn, certificate.subject);
225                        Assert.assertEquals(dn, certificate.issuer);
226                    }
227                }
228            }
229        });
230        Main.main(argsList.toArray(new String[]{}));
231        assert false;
232    }
