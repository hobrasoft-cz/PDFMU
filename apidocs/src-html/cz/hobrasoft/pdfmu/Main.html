078    private static ArgumentParser createBasicParser() {
079        // Create a command line argument parser
080        ArgumentParser parser = ArgumentParsers.newArgumentParser("pdfmu")
081                .description("PDF Manipulation Utility")
082                .defaultHelp(true);
083
084        parser.version(getProjectVersion());
085        parser.addArgument("-v", "--version")
086                .help("show version and exit")
087                .action(Arguments.version());
088
089        parser.addArgument("--legal-notice")
090                .help("show legal notice and exit")
091                .action(new PrintAndExitAction(getLegalNotice()));
092
093        // TODO: Use an enum
094        parser.addArgument("--output-format")
095                .choices("text", "json")
096                .setDefault("text")
097                .type(String.class)
098                .help("format of stderr output");
099
100        return parser;
101    }

103    private static final String POM_PROPERTIES_RESOURCE_NAME = "pom.properties";
104    private static final Properties POM_PROPERTIES = new Properties();
105
106    private static void loadPomProperties() {
107        ClassLoader classLoader = Main.class.getClassLoader();
108        InputStream in = classLoader.getResourceAsStream(POM_PROPERTIES_RESOURCE_NAME);
109        if (in != null) {
110            try {
111                POM_PROPERTIES.load(in);
112            } catch (IOException ex) {
113                logger.severe(String.format("Could not load the POM properties file: %s", ex));
114            }
115            try {
116                in.close();
117            } catch (IOException ex) {
118                logger.severe(String.format("Could not close the POM properties file: %s", ex));
119            }
120        } else {
121            logger.severe("Could not open the POM properties file.");
122        }
123    }

165    /**
166     * Use {@link LinkedHashMap} or {@link java.util.TreeMap} as operations to
167     * specify the order in which the operations are printed.
168     *
169     * @param operations a map that assigns the operations to their names
170     * @return an argument parser with the operations attached as sub-commands
171     */
172    private static ArgumentParser createFullParser(Map<String, Operation> operations) {
173        // Create a command line argument parser
174        ArgumentParser parser = createBasicParser();
175
176        // Create a Subparsers instance for operation subparsers
177        Subparsers subparsers = parser.addSubparsers()
178                .title("operations")
179                .metavar("OPERATION")
180                .help("operation to execute")
181                .dest("operation");
182
183        // Configure operation subparsers
184        for (Map.Entry<String, Operation> e : operations.entrySet()) {
185            String name = e.getKey();
186            Operation operation = e.getValue();
187            operation.configureSubparser(subparsers.addParser(name));
188        }
189
190        return parser;
191    }

193    private static OperationException apeToOe(ArgumentParserException e) {
194        Set<ExceptionMessagePattern> patterns = new HashSet<>();
195
196        patterns.add(new ExceptionMessagePattern(INPUT_NOT_FOUND,
197                "argument (?<argument>.*): Insufficient permissions to read file: \'(?<file>.*)\'",
198                Arrays.asList(new String[]{"argument", "file"})));
199
200        // UnrecognizedArgumentException
201        patterns.add(new ExceptionMessagePattern(PARSER_UNRECOGNIZED_ARGUMENT,
202                "unrecognized arguments: '(?<argument>.*)'",
203                Arrays.asList(new String[]{"argument"})));
204
205        // ArgumentParserException
206        patterns.add(new ExceptionMessagePattern(PARSER_INVALID_CHOICE,
207                "argument (?<argument>.*): invalid choice: '(?<choice>.*)' \\(choose from \\{(?<validChoices>.*)\\}\\)",
208                Arrays.asList(new String[]{"argument", "choice", "validChoices"})));
209
210        // UnrecognizedCommandException
211        patterns.add(new ExceptionMessagePattern(PARSER_UNRECOGNIZED_COMMAND,
212                "invalid choice: '(?<command>.*)' \\(choose from (?<validCommands>.*)\\)",
213                Arrays.asList(new String[]{"command", "validCommands"})));
214
215        // ArgumentParserException
216        patterns.add(new ExceptionMessagePattern(PARSER_TOO_FEW_ARGUMENTS,
217                "too few arguments",
218                new ArrayList<String>()));
219
220        OperationException oe = null;
221        for (ExceptionMessagePattern p : patterns) {
222            oe = p.getOperationException(e);
223            if (oe != null) {
224                break;
225            }
226        }
227
228        if (oe == null) {
229            // Unknown parser exception
230            oe = new OperationException(PARSER_UNKNOWN, e);
231        }
232
233        assert oe != null;
234        return oe;
235    }

237    /**
238     * The main entry point of PDFMU
239     *
240     * @param args the command line arguments
241     */
242    public static void main(String[] args) {
243        int exitStatus = 0; // Default: 0 (normal termination)
244
245        // Create a map of operations
246        Map<String, Operation> operations = new LinkedHashMap<>();
247        operations.put("inspect", OperationInspect.getInstance());
248        operations.put("update-version", OperationVersionSet.getInstance());
249        operations.put("update-properties", OperationMetadataSet.getInstance());
250        operations.put("attach", OperationAttach.getInstance());
251        operations.put("sign", OperationSignatureAdd.getInstance());
252
253        // Create a command line argument parser
254        ArgumentParser parser = createFullParser(operations);
255
256        // Parse command line arguments
257        Namespace namespace = null;
258        try {
259            // If help is requested,
260            // `parseArgs` prints help message and throws `ArgumentParserException`
261            // (so `namespace` stays null).
262            // If insufficient or invalid `args` are given,
263            // `parseArgs` throws `ArgumentParserException`.
264            namespace = parser.parseArgs(args);
265        } catch (HelpScreenException e) {
266            parser.handleError(e); // Do nothing
267        } catch (UnrecognizedCommandException e) {
268            exitStatus = PARSER_UNRECOGNIZED_COMMAND.getCode();
269            parser.handleError(e); // Print the error in human-readable format
270        } catch (UnrecognizedArgumentException e) {
271            exitStatus = PARSER_UNRECOGNIZED_ARGUMENT.getCode();
272            parser.handleError(e); // Print the error in human-readable format
273        } catch (ArgumentParserException ape) {
274            OperationException oe = apeToOe(ape);
275            exitStatus = oe.getCode();
276            // We could also write `oe` as a JSON document,
277            // but we do not know whether JSON output was requested,
278            // so we use the text output (default).
279
280            parser.handleError(ape); // Print the error in human-readable format
281        }
282
283        if (namespace == null) {
284            System.exit(exitStatus);
285        }
286
287        assert exitStatus == 0;
288
289        // Handle command line arguments
290        WritingMapper wm = null;
291
292        // Extract operation name
293        String operationName = namespace.getString("operation");
294        assert operationName != null; // The argument "operation" is a sub-command, thus it is required
295
296        // Select the operation from `operations`
297        assert operations.containsKey(operationName); // Only supported operation names are allowed
298        Operation operation = operations.get(operationName);
299        assert operation != null;
300
301        // Choose the output format
302        String outputFormat = namespace.getString("output_format");
303        switch (outputFormat) {
304            case "json":
305                // Disable loggers
306                disableLoggers();
307                // Initialize the JSON serializer
308                wm = new WritingMapper();
309                operation.setWritingMapper(wm); // Configure the operation
310                break;
311            case "text":
312                // Initialize the text output
313                TextOutput to = new TextOutput(System.err); // Bind to `System.err`
314                operation.setTextOutput(to); // Configure the operation
315                break;
316            default:
317                assert false; // The option has limited choices
318        }
319
320        // Execute the operation
321        try {
322            operation.execute(namespace);
323        } catch (OperationException ex) {
324            exitStatus = ex.getCode();
325
326            // Log the exception
327            logger.severe(ex.getLocalizedMessage());
328            Throwable cause = ex.getCause();
329            if (cause != null && cause.getMessage() != null) {
330                logger.severe(cause.getLocalizedMessage());
331            }
332
333            if (wm != null) {
334                // JSON output is enabled
335                ex.writeInWritingMapper(wm);
336            }
337        }
338        System.exit(exitStatus);
339    }
